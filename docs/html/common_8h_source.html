86inline double sind(double x) {
87 if (!la::isfinite(x)) return sin(x);
88 if (x < 0.0) return -sind(-x);
89 int quo;
90 x = remquo(fabs(x), 90.0, &quo);
91 switch (quo % 4) {
92 case 0:
93 return sin(radians(x));
94 case 1:
95 return cos(radians(x));
96 case 2:
97 return -sin(radians(x));
98 case 3:
99 return -cos(radians(x));
100 }
101 return 0.0;
102}

480class Quality {
481 private:
482 inline static int circularSegments_ = DEFAULT_SEGMENTS;
483 inline static double circularAngle_ = DEFAULT_ANGLE;
484 inline static double circularEdgeLength_ = DEFAULT_LENGTH;
485
486 public:
497 static void SetMinCircularAngle(double angle) {
498 if (angle <= 0) return;
499 circularAngle_ = angle;
500 }
501
511 static void SetMinCircularEdgeLength(double length) {
512 if (length <= 0) return;
513 circularEdgeLength_ = length;
514 }
515
525 static void SetCircularSegments(int number) {
526 if (number < 3 && number != 0) return;
527 circularSegments_ = number;
528 }
529
537 static int GetCircularSegments(double radius) {
538 if (circularSegments_ > 0) return circularSegments_;
539 int nSegA = 360.0 / circularAngle_;
540 int nSegL = 2.0 * radius * kPi / circularEdgeLength_;
541 int nSeg = fmin(nSegA, nSegL) + 3;
542 nSeg -= nSeg % 4;
543 return std::max(nSeg, 3);
544 }

551 static void ResetToDefaults() {
552 circularSegments_ = DEFAULT_SEGMENTS;
553 circularAngle_ = DEFAULT_ANGLE;
554 circularEdgeLength_ = DEFAULT_LENGTH;
555 }
