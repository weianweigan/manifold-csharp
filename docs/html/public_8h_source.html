79inline double sind(double x) {
80 if (!std::isfinite(x)) return sin(x);
81 if (x < 0.0) return -sind(-x);
82 int quo;
83 x = remquo(fabs(x), 90.0, &quo);
84 switch (quo % 4) {
85 case 0:
86 return sin(glm::radians(x));
87 case 1:
88 return cos(glm::radians(x));
89 case 2:
90 return -sin(glm::radians(x));
91 case 3:
92 return -cos(glm::radians(x));
93 }
94 return 0.0;
95}

110inline mat4x3 RotateUp(vec3 up) {
111 up = glm::normalize(up);
112 vec3 axis = glm::cross(up, {0, 0, 1});
113 double angle = glm::asin(glm::length(axis));
114 if (glm::dot(up, {0, 0, 1}) < 0) angle = glm::pi<double>() - angle;
115 return mat4x3(glm::rotate(mat4(1), angle, axis));
116}

129inline int CCW(vec2 p0, vec2 p1, vec2 p2, double tol) {
130 vec2 v1 = p1 - p0;
131 vec2 v2 = p2 - p0;
132 double area = fma(v1.x, v2.y, -v1.y * v2.x);
133 double base2 = glm::max(glm::dot(v1, v1), glm::dot(v2, v2));
134 if (area * area * 4 <= base2 * tol * tol)
135 return 0;
136 else
137 return area > 0 ? 1 : -1;
138}

535class Quality {
536 private:
537 inline static int circularSegments_ = DEFAULT_SEGMENTS;
538 inline static double circularAngle_ = DEFAULT_ANGLE;
539 inline static double circularEdgeLength_ = DEFAULT_LENGTH;
540
541 public:
552 static void SetMinCircularAngle(double angle) {
553 if (angle <= 0) return;
554 circularAngle_ = angle;
555 }
556
566 static void SetMinCircularEdgeLength(double length) {
567 if (length <= 0) return;
568 circularEdgeLength_ = length;
569 }
570
580 static void SetCircularSegments(int number) {
581 if (number < 3 && number != 0) return;
582 circularSegments_ = number;
583 }
584
592 static int GetCircularSegments(double radius) {
593 if (circularSegments_ > 0) return circularSegments_;
594 int nSegA = 360.0 / circularAngle_;
595 int nSegL = 2.0 * radius * glm::pi<double>() / circularEdgeLength_;
596 int nSeg = fmin(nSegA, nSegL) + 3;
597 nSeg -= nSeg % 4;
598 return std::max(nSeg, 3);
599 }
600
601 static void ResetToDefaults() {
602 circularSegments_ = DEFAULT_SEGMENTS;
603 circularAngle_ = DEFAULT_ANGLE;
604 circularEdgeLength_ = DEFAULT_LENGTH;
605 }
606};
